function workoutResult($client_id){

        $db = new Database();
        $conn = $db->getConnection();
        $stmt = $conn -> prepare("SELECT select es.sheet_name, rd.result, rd.date
                                  FROM exercise_sheet es, result_details rd
                                  WHERE es.sheet_id = rd.sheet_id
                                  AND rd.client_id = ?
                                  ORDER BY rd.date desc");
        $stmt->bind_param("i",$sheet_id);
        $stmt->execute();
        $result=$stmt->get_result();
        $stmt->close();

        $history = array();

        while($details = $result->fetch_assoc()){
            array_push($history,$details);
        }

        return $history;

}
